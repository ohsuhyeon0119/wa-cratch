```typescript
import type { RefObject } from 'react'
import * as Blockly from 'blockly'
import type { FunctionDeclaration } from '@google/genai'

export const GET_PROJECT_CONTEXT_DECL: FunctionDeclaration = {
  name: 'get_project_context',
  description: '현재 열린 프로젝트의 제목, 블록 수, 블록 구조 JSON을 반환합니다.',
  parametersJsonSchema: {
    type: 'object',
    properties: {},
  },
}

export function executeGetProjectContext(
  workspaceRef: RefObject<Blockly.WorkspaceSvg | null>,
  projectTitle: string
): Record<string, unknown> {
  const workspace = workspaceRef.current
  if (!workspace) return { title: projectTitle, blockCount: 0, blocks_json: {} }
  return {
    title: projectTitle,
    blockCount: workspace.getAllBlocks(false).length,
    blocks_json: Blockly.serialization.workspaces.save(workspace),
  }
}
```

```typescript
// frontend/src/components/VoiceAgent/useGeminiLive.ts
import { useState, useRef, useCallback, useEffect } from 'react'
import type { RefObject } from 'react'
import * as Blockly from 'blockly'
import { GoogleGenAI } from '@google/genai'
import type { LiveServerMessage, Session } from '@google/genai'
import { GET_PROJECT_CONTEXT_DECL, executeGetProjectContext } from './tools'

export type VoiceState = 'idle' | 'connecting' | 'listening' | 'speaking'
export type Message = { role: 'user' | 'model'; text: string; ts: number }

const SYSTEM_PROMPT = `너는 WaCratch 블록 코딩 에디터의 AI 도우미야.
어린이 사용자가 블록 코딩을 배울 수 있도록 친절하고 쉽게 설명해줘.
get_project_context 툴로 현재 프로젝트 상태를 확인할 수 있어.
한국어로 대답해.`

const WORKLET_CODE = `
class AudioCaptureWorklet extends AudioWorkletProcessor {
  buffer = new Int16Array(2048)
  writeIndex = 0
  process(inputs) {
    const channel = inputs[0]?.[0]
    if (!channel) return true
    for (let i = 0; i < channel.length; i++) {
      this.buffer[this.writeIndex++] = Math.max(-32768, Math.min(32767, channel[i] * 32768))
      if (this.writeIndex >= this.buffer.length) {
        this.port.postMessage({ buffer: this.buffer.slice(0, this.writeIndex).buffer })
        this.writeIndex = 0
      }
    }
    return true
  }
}
registerProcessor('audio-capture-worklet', AudioCaptureWorklet)
`

function base64ToFloat32(base64: string): Float32Array {
  const binary = atob(base64)
  const bytes = new Uint8Array(binary.length)
  for (let i = 0; i < binary.length; i++) bytes[i] = binary.charCodeAt(i)
  const samples = new Int16Array(bytes.buffer)
  const out = new Float32Array(samples.length)
  for (let i = 0; i < samples.length; i++) out[i] = samples[i] / 32768
  return out
}

export function useGeminiLive(
  workspaceRef: RefObject<Blockly.WorkspaceSvg | null>,
  projectTitle: string
) {
  const [voiceState, setVoiceState] = useState<VoiceState>('idle')
  const [messages, setMessages] = useState<Message[]>([])

  const sessionRef = useRef<Session | null>(null)
  const mediaStreamRef = useRef<MediaStream | null>(null)
  const inputCtxRef = useRef<AudioContext | null>(null)
  const outputCtxRef = useRef<AudioContext | null>(null)
  const nextPlayTimeRef = useRef(0)

  // projectTitle 변경에 반응하기 위해 ref로 관리
  const projectTitleRef = useRef(projectTitle)
  useEffect(() => { projectTitleRef.current = projectTitle }, [projectTitle])

  const addMessage = useCallback((role: 'user' | 'model', text: string) => {
    setMessages(prev => [...prev, { role, text, ts: Date.now() }])
  }, [])

  const playAudio = useCallback((base64: string) => {
    const ctx = outputCtxRef.current
    if (!ctx) return
    const float32 = base64ToFloat32(base64)
    const buf = ctx.createBuffer(1, float32.length, 24000)
    buf.copyToChannel(float32, 0)
    const src = ctx.createBufferSource()
    src.buffer = buf
    src.connect(ctx.destination)
    const now = ctx.currentTime
    if (nextPlayTimeRef.current < now) nextPlayTimeRef.current = now
    src.start(nextPlayTimeRef.current)
    nextPlayTimeRef.current += buf.duration
  }, [])

  const handleMessageRef = useRef<(e: LiveServerMessage) => void>(() => {})

  const handleMessage = useCallback((e: LiveServerMessage) => {
    const inputText = e.serverContent?.inputTranscription?.text?.trim()
    if (inputText) addMessage('user', inputText)

    const outputText = e.serverContent?.outputTranscription?.text?.trim()
    if (outputText) addMessage('model', outputText)

    const parts = e.serverContent?.modelTurn?.parts ?? []
    for (const part of parts) {
      if (part.inlineData?.data) {
        setVoiceState('speaking')
        playAudio(part.inlineData.data)
      }
    }

    if (e.serverContent?.turnComplete) setVoiceState('listening')

    for (const call of e.toolCall?.functionCalls ?? []) {
      if (call.name === 'get_project_context' && call.id) {
        const result = executeGetProjectContext(workspaceRef, projectTitleRef.current)
        sessionRef.current?.sendToolResponse({
          functionResponses: [{ id: call.id, response: result }],
        })
      }
    }
  }, [addMessage, playAudio, workspaceRef])

  useEffect(() => { handleMessageRef.current = handleMessage }, [handleMessage])

  function cleanup() {
    mediaStreamRef.current?.getTracks().forEach(t => t.stop())
    mediaStreamRef.current = null
    inputCtxRef.current?.close()
    inputCtxRef.current = null
    outputCtxRef.current?.close()
    outputCtxRef.current = null
  }

  const connect = useCallback(async () => {
    setVoiceState('connecting')

    outputCtxRef.current = new AudioContext({ sampleRate: 24000 })
    nextPlayTimeRef.current = 0

    const stream = await navigator.mediaDevices.getUserMedia({ audio: true })
    mediaStreamRef.current = stream

    const inputCtx = new AudioContext({ sampleRate: 16000 })
    inputCtxRef.current = inputCtx

    const blobUrl = URL.createObjectURL(new Blob([WORKLET_CODE], { type: 'application/javascript' }))
    await inputCtx.audioWorklet.addModule(blobUrl)
    URL.revokeObjectURL(blobUrl)

    const micSource = inputCtx.createMediaStreamSource(stream)
    const workletNode = new AudioWorkletNode(inputCtx, 'audio-capture-worklet')
    micSource.connect(workletNode)

    const ai = new GoogleGenAI({ apiKey: import.meta.env.VITE_GEMINI_API_KEY })
    const session = await ai.live.connect({
      model: 'gemini-2.0-flash-live-preview',
      config: {
        responseModalities: ['AUDIO'],
        inputAudioTranscription: {},
        outputAudioTranscription: {},
        systemInstruction: { parts: [{ text: SYSTEM_PROMPT }] },
        tools: [{ functionDeclarations: [GET_PROJECT_CONTEXT_DECL] }],
      },
      callbacks: {
        onopen: () => setVoiceState('listening'),
        onmessage: (e) => handleMessageRef.current(e),
        onclose: () => { setVoiceState('idle'); cleanup() },
        onerror: () => { setVoiceState('idle'); cleanup() },
      },
    })
    sessionRef.current = session

    workletNode.port.onmessage = (ev: MessageEvent<{ buffer: ArrayBuffer }>) => {
      session.sendRealtimeInput({
        media: new Blob([ev.data.buffer], { type: 'audio/pcm;rate=16000' }),
      })
    }
  }, [])

  const disconnect = useCallback(() => {
    sessionRef.current?.close()
    sessionRef.current = null
    cleanup()
    setVoiceState('idle')
  }, [])

  return { voiceState, messages, connect, disconnect }
}
```

```tsx
// frontend/src/components/VoiceAgent/VoiceAgentPanel.tsx
import { useEffect, useRef } from 'react'
import type { VoiceState, Message } from './useGeminiLive'
import s from './VoiceAgent.module.css'

const STATE_LABELS: Record<VoiceState, string> = {
  idle: '대기 중',
  connecting: '연결 중...',
  listening: '듣고 있어요 🎤',
  speaking: '말하는 중 💬',
}

function formatTime(ts: number): string {
  return new Date(ts).toLocaleTimeString('ko-KR', { hour: '2-digit', minute: '2-digit' })
}

interface Props {
  open: boolean
  voiceState: VoiceState
  messages: Message[]
  onClose: () => void
}

export default function VoiceAgentPanel({ open, voiceState, messages, onClose }: Props) {
  const bottomRef = useRef<HTMLDivElement>(null)

  useEffect(() => {
    bottomRef.current?.scrollIntoView({ behavior: 'smooth' })
  }, [messages])

  return (
    <div className={`${s.panel} ${open ? '' : s.panelHidden}`}>
      <div className={s.panelHeader}>
        <div>
          <div className={s.panelTitle}>🎙️ AI 와냥이</div>
          <div className={s.panelStatus}>{STATE_LABELS[voiceState]}</div>
        </div>
        <button className={s.closeBtn} onClick={onClose} aria-label="닫기">✕</button>
      </div>

      <div className={s.chatArea}>
        {messages.length === 0 ? (
          <div className={s.emptyState}>
            패널을 열면 대화가 시작돼요!<br />무엇이든 물어보세요 🐾
          </div>
        ) : (
          messages.map((msg) => (
            <div
              key={msg.ts}
              className={`${s.message} ${msg.role === 'user' ? s.messageUser : s.messageModel}`}
            >
              <div className={s.bubble}>{msg.text}</div>
              <div className={s.timestamp}>{formatTime(msg.ts)}</div>
            </div>
          ))
        )}
        <div ref={bottomRef} />
      </div>
    </div>
  )
}
```

```tsx
// frontend/src/components/VoiceAgent/VoiceAgent.tsx
import { useState, useCallback } from 'react'
import * as Blockly from 'blockly'
import { useGeminiLive } from './useGeminiLive'
import VoiceAgentToggle from './VoiceAgentToggle'
import VoiceAgentPanel from './VoiceAgentPanel'

interface Props {
  workspaceRef: RefObject<Blockly.WorkspaceSvg | null>
  projectTitle: string
}

export default function VoiceAgent({ workspaceRef, projectTitle }: Props) {
  const [open, setOpen] = useState(false)
  const { voiceState, messages, connect, disconnect } = useGeminiLive(workspaceRef, projectTitle)

  const handleToggle = useCallback(() => {
    if (open) {
      disconnect()
      setOpen(false)
    } else {
      setOpen(true)
      connect().catch(console.error)
    }
  }, [open, connect, disconnect])

  const handleClose = useCallback(() => {
    disconnect()
    setOpen(false)
  }, [disconnect])

  return (
    <>
      <VoiceAgentToggle voiceState={voiceState} onClick={handleToggle} />
      <VoiceAgentPanel
        open={open}
        voiceState={voiceState}
        messages={messages}
        onClose={handleClose}
      />
    </>
  )
}
```
